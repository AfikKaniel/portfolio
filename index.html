<script>

const items=document.querySelectorAll('.timeline-item'),

sections=document.querySelectorAll('.milestone'),

fireProgress=document.getElementById('fireProgress'),

fireParticles=document.getElementById('fireParticles'),

fireGlow=document.getElementById('fireGlow');


function updateTimeline(){

const scrollPos=window.scrollY+window.innerHeight/2;

let activeIdx=-1;

sections.forEach((s,i)=>{if(scrollPos>=s.offsetTop)activeIdx=i});


// Only activate items and fire when user reaches Playtika section

const reachedMilestones = activeIdx >= 0;

items.forEach((it,i)=>{it.classList.toggle('active',reachedMilestones && i<=activeIdx)});


// Calculate flame position based on icon centers

let pct = 0;

if(reachedMilestones && activeIdx >= 0){

// If at the last milestone (amp), fill 100%

if(activeIdx === items.length - 1){

pct = 100;

}else{

const timeline = document.querySelector('.timeline');

const timelineRect = timeline.getBoundingClientRect();

const timelineWidth = timelineRect.width;

const activeItem = items[activeIdx];

const itemRect = activeItem.getBoundingClientRect();

// Get center of the icon (where white circle touches timeline)

const iconCenter = itemRect.left + (itemRect.width / 2) - timelineRect.left;

pct = (iconCenter / timelineWidth) * 100;

}

}


// Fire only appears when user reaches first milestone

if(reachedMilestones){

fireProgress.style.width=pct+'%';

fireParticles.style.width=pct+'%';

fireGlow.style.width=pct+'%';

fireProgress.style.opacity='1';

fireParticles.style.opacity='1';

fireGlow.style.opacity='1';

}else{

fireProgress.style.width='0%';

fireParticles.style.width='0%';

fireGlow.style.width='0%';

fireProgress.style.opacity='0';

fireParticles.style.opacity='0';

fireGlow.style.opacity='0';

}

}


items.forEach(it=>{

it.addEventListener('click',()=>{

const target=document.getElementById(it.dataset.target);

if(target)target.scrollIntoView({behavior:'smooth',block:'start'});

});

});


window.addEventListener('scroll',updateTimeline);

window.addEventListener('resize',updateTimeline);

updateTimeline();


// Auto-play videos when they come into view

const videoObserver = new IntersectionObserver((entries) => {

entries.forEach(entry => {

const video = entry.target;

if(entry.isIntersecting){

video.play();

}else{

video.pause();

}

});

}, {threshold: 0.5});


document.querySelectorAll('.media-item video').forEach(video => {

video.muted = true;

video.loop = true;

video.playsInline = true;

videoObserver.observe(video);

});

</script>
